  async createProtocolVersion(data: CreateProtocolVersionRequest): Promise<ProtocolVersion> {
    const { data: result, error } = await this.supabase
      .from('protocol_versions')
      .insert({
        ...data,
        created_by: (await this.supabase.auth.getUser()).data.user?.id,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      })
      .select()
      .single();

    if (error) throw new Error(`Failed to create protocol version: ${error.message}`);
    return result;
  }

  async getProtocolVersions(filters?: {
    protocol_name?: string;
    approval_status?: string;
    is_active?: boolean;
    page?: number;
    per_page?: number;
  }): Promise<{ data: ProtocolVersionWithAnalytics[]; total: number }> {
    let query = this.supabase
      .from('protocol_versions')
      .select(`
        *,
        latest_analytics:protocol_analytics(*)
      `);

    if (filters?.protocol_name) {
      query = query.ilike('protocol_name', `%${filters.protocol_name}%`);
    }
    if (filters?.approval_status) {
      query = query.eq('approval_status', filters.approval_status);
    }
    if (filters?.is_active !== undefined) {
      query = query.eq('is_active', filters.is_active);
    }

    const page = filters?.page || 1;
    const per_page = filters?.per_page || 20;
    const from = (page - 1) * per_page;
    const to = from + per_page - 1;

    const { data, error, count } = await query
      .range(from, to)
      .order('created_at', { ascending: false });

    if (error) throw new Error(`Failed to fetch protocol versions: ${error.message}`);

    // Add outcome and feedback counts for each protocol
    const protocolsWithCounts = await Promise.all(
      (data || []).map(async (protocol) => {
        const [outcomeCount, feedbackCount] = await Promise.all([
          this.getProtocolOutcomeCount(protocol.id),
          this.getProtocolFeedbackCount(protocol.id),
        ]);

        return {
          ...protocol,
          outcome_count: outcomeCount,
          feedback_count: feedbackCount,
        };
      })
    );

    return { data: protocolsWithCounts, total: count || 0 };
  }

  async getProtocolOutcomes(filters?: {
    protocol_version_id?: string;
    patient_id?: string;
    success_score_min?: number;
    success_score_max?: number;
    page?: number;
    per_page?: number;

      query = query.eq('patient_id', filters.patient_id);

    if (filters?.success_score_min !== undefined) {
      query = query.gte('success_score', filters.success_score_min);
    }
    if (filters?.success_score_max !== undefined) {
      query = query.lte('success_score', filters.success_score_max);
    }

    const page = filters?.page || 1;
    const per_page = filters?.per_page || 20;
    const from = (page - 1) * per_page;
    const to = from + per_page - 1;

    const { data, error, count } = await query
      .range(from, to)
      .order('outcome_date', { ascending: false });

    if (error) throw new Error(`Failed to fetch protocol outcomes: ${error.message}`);
    return { data: data || [], total: count || 0 };
  }

  // Optimization Analysis Engine
  async triggerOptimizationAnalysis(protocol_id: string): Promise<OptimizationResult> {
    // Get protocol data and recent outcomes
    const [protocol, outcomes, feedback] = await Promise.all([
      this.getProtocolVersionById(protocol_id),
      this.getProtocolOutcomes({ protocol_version_id: protocol_id }),
      this.getProtocolFeedback({ protocol_id }),
    ]);

    if (!protocol) {
      throw new Error('Protocol not found for optimization analysis');
    }

    // Perform automated analysis
    const analysis = await this.performOptimizationAnalysis(protocol, outcomes.data, feedback.data);

    // Store optimization results
    const { data: result, error } = await this.supabase
      .from('optimization_results')
      .insert({
        protocol_id,
        optimization_type: 'automated_analysis',
        improvement_metrics: analysis.metrics,

        confidence_score: analysis.confidence,
        statistical_significance: analysis.significance,
        recommendation_data: analysis.recommendations,
        automated_analysis: analysis.full_analysis,
        human_review_required: analysis.confidence < 80,
        created_at: new Date().toISOString(),
        updated_at: new Date().toISOString(),
      })
      .select()
      .single();

    if (error) throw new Error(`Failed to store optimization results: ${error.message}`);
    return result;

  private async performOptimizationAnalysis(

      Array.isArray(o.complications) && o.complications.length > 0
    ).length / outcomes.length * 100;

    // Analyze feedback patterns

    const improvementSuggestions = feedback.filter(f => 
      f.feedback_type === 'improvement' || f.feedback_type === 'suggestion'
    ).length;

    // Generate recommendations based on analysis
    const recommendations = [];
    
    if (successRate < 75) {
      recommendations.push({
        type: 'success_rate_improvement',
        priority: 'high',
        suggestion: 'Consider reviewing protocol steps with low success rates',
        potential_improvement: 15,
      });
    }

    if (complicationRate > 10) {
      recommendations.push({
        type: 'complication_reduction',
        priority: 'critical',
        suggestion: 'Implement additional safety measures to reduce complications',
        potential_improvement: 20,
      });
    }

    if (averageSatisfaction < 7) {
      recommendations.push({
        type: 'satisfaction_improvement',
        priority: 'medium',
        suggestion: 'Review patient communication and expectations management',
        potential_improvement: 10,
      });
    }

    // Calculate confidence score based on data quality and quantity
    const confidence = Math.min(100, 
      (outcomes.length * 2) + 
      (feedback.length * 1.5) + 
      (protocol.optimization_data ? 10 : 0)
    );

    // Calculate statistical significance (simplified)
    const significance = outcomes.length >= 30 ? 0.05 : 0.1;

    return {
      metrics: {
        success_rate: successRate,
        average_satisfaction: averageSatisfaction,
        complication_rate: complicationRate,
        total_outcomes: outcomes.length,
        total_feedback: feedback.length,
        critical_feedback_count: criticalFeedback,
      },
      validation: {
        sample_size: outcomes.length,
        feedback_quality: feedback.length > 0 ? 'adequate' : 'limited',
        data_completeness: outcomes.length > 10 ? 'good' : 'limited',
      },
      confidence,
      significance,
      recommendations: {
        high_priority: recommendations.filter(r => r.priority === 'critical' || r.priority === 'high'),
        medium_priority: recommendations.filter(r => r.priority === 'medium'),
        all_recommendations: recommendations,
      },
      full_analysis: {
        analyzed_at: new Date().toISOString(),
        protocol_version: protocol.version_number,
        analysis_scope: {
          outcomes_period: outcomes.length > 0 ? {
            start: Math.min(...outcomes.map(o => new Date(o.outcome_date).getTime())),
            end: Math.max(...outcomes.map(o => new Date(o.outcome_date).getTime())),
          } : null,
          feedback_summary: {
            total: feedback.length,
            by_type: feedback.reduce((acc, f) => {

  async analyzeExperimentResults(experiment_id: string): Promise<ProtocolExperimentResults> {
    const { data: experiment, error } = await this.supabase
      .from('protocol_experiments')
      .select('*')
      .eq('id', experiment_id)
      .single();

    if (error) throw new Error(`Failed to fetch experiment: ${error.message}`);

    // Get outcomes for both control and test protocols
    const [controlOutcomes, testOutcomes] = await Promise.all([
      experiment.control_protocol_id 
        ? this.getProtocolOutcomes({ protocol_version_id: experiment.control_protocol_id })
        : { data: [], total: 0 },
      experiment.test_protocol_id 
        ? this.getProtocolOutcomes({ protocol_version_id: experiment.test_protocol_id })
        : { data: [], total: 0 },
    ]);

    // Perform statistical analysis
    const controlMean = controlOutcomes.data.length > 0 
      ? (controlOutcomes.data as any[]).reduce((sum, o) => sum + o.success_score, 0) / controlOutcomes.data.length
      : 0;

    const testMean = testOutcomes.data.length > 0 
      ? (testOutcomes.data as any[]).reduce((sum, o) => sum + o.success_score, 0) / testOutcomes.data.length
      : 0;

    const effectSize = testMean - controlMean;
    const pooledSampleSize = controlOutcomes.data.length + testOutcomes.data.length;
    
    // Simplified statistical calculations
    const standardError = Math.sqrt((controlMean * (100 - controlMean) / controlOutcomes.data.length) + 
                                   (testMean * (100 - testMean) / testOutcomes.data.length));
    
    const zScore = standardError > 0 ? effectSize / standardError : 0;
    const pValue = Math.max(0.001, 1 - Math.abs(zScore) / 2); // Simplified p-value calculation
    
    const confidenceInterval: [number, number] = [
      effectSize - 1.96 * standardError,
      effectSize + 1.96 * standardError,
    ];

    const isSignificant = pValue < experiment.significance_level;
    const winner = effectSize > 0 ? 'test' : 'control';

    return {
      experiment,
      statistical_analysis: {
        p_value: pValue,
        confidence_interval: confidenceInterval,
        effect_size: effectSize,
        statistical_power: experiment.statistical_power,
      },
      recommendations: {
        winner: isSignificant ? winner : undefined,
        significance: isSignificant,
        next_steps: isSignificant 
          ? [`Implement ${winner} protocol`, 'Monitor performance', 'Plan rollout strategy']
          : ['Continue experiment', 'Increase sample size', 'Review protocol differences'],
      },
    };
  }

  // Analytics and Reporting
  async generateProtocolAnalytics(request: ProtocolAnalyticsRequest): Promise<ProtocolAnalytics> {
    const { protocol_version_id, analytics_period, period_start, period_end } = request;

    // Get outcomes for the specified period
    let outcomeQuery = this.supabase
      .from('protocol_outcomes')
      .select('*')
      .gte('outcome_date', period_start)
      .lte('outcome_date', period_end);

    if (protocol_version_id) {
      outcomeQuery = outcomeQuery.eq('protocol_version_id', protocol_version_id);
    }

    const { data: outcomes, error: outcomeError } = await outcomeQuery;

    if (outcomeError) throw new Error(`Failed to fetch outcomes for analytics: ${outcomeError.message}`);

    // Calculate analytics metrics
    const usageCount = outcomes?.length || 0;
    const successRate = usageCount > 0 
      ? outcomes!.reduce((sum, o) => sum + o.success_score, 0) / usageCount
      : 0;

    const averageOutcomeScore = successRate;
    const patientSatisfaction = outcomes?.filter(o => o.patient_satisfaction)
      .reduce((sum, o) => sum + (o.patient_satisfaction || 0), 0) / 
      (outcomes?.filter(o => o.patient_satisfaction).length || 1);

    const complicationRate = usageCount > 0 
      ? (outcomes!.filter(o => Array.isArray(o.complications) && o.complications.length > 0).length / usageCount) * 100
      : 0;

    // Create analytics record
    const analyticsData = {
      protocol_version_id: protocol_version_id || 'all_protocols',
      analytics_period,
      period_start,
      period_end,
      usage_count: usageCount,
      success_rate: successRate,
      average_outcome_score: averageOutcomeScore,
      patient_satisfaction: patientSatisfaction,
      complication_rate: complicationRate,
      performance_metrics: {
        total_treatments: usageCount,
        successful_treatments: outcomes?.filter(o => o.success_score >= 75).length || 0,
        complications: outcomes?.filter(o => Array.isArray(o.complications) && o.complications.length > 0).length || 0,
        side_effects: outcomes?.filter(o => Array.isArray(o.side_effects) && o.side_effects.length > 0).length || 0,
      },
      calculated_at: new Date().toISOString(),
      created_at: new Date().toISOString(),
    };

    const { data: result, error } = await this.supabase
      .from('protocol_analytics')
      .insert(analyticsData)
      .select()
      .single();

    if (error) throw new Error(`Failed to create protocol analytics: ${error.message}`);
    return result;
  }
