### Database Schema Integration

#### LGPD-Specific Tables
```sql
-- Consent Records
CREATE TABLE lgpd_consent_records (
  id UUID PRIMARY KEY,
  clinic_id UUID REFERENCES clinics(id),
  user_id UUID REFERENCES users(id),
  data_types TEXT[],
  purposes TEXT[],
  consent_text TEXT,
  version INTEGER,
  status consent_status,
  collected_at TIMESTAMP,
  expires_at TIMESTAMP,
  withdrawn_at TIMESTAMP,
  metadata JSONB
);

-- Audit Trail
CREATE TABLE lgpd_audit_records (
  id UUID PRIMARY KEY,
  clinic_id UUID REFERENCES clinics(id),
  event_type lgpd_audit_event_type,
  severity lgpd_audit_severity,
  data_type lgpd_data_type,
  description TEXT,
  user_id UUID,
  timestamp TIMESTAMP,
  compliance_status TEXT,
  metadata JSONB
);

-- Data Retention
CREATE TABLE lgpd_data_retention_records (
  id UUID PRIMARY KEY,
  clinic_id UUID REFERENCES clinics(id),
  data_type lgpd_data_type,
  data_id TEXT,
  collected_at TIMESTAMP,
  retention_expires_at TIMESTAMP,
  status retention_status,
  policy_id UUID,
  metadata JSONB
);
```

#### RLS Policies
```sql
-- Clinic-based isolation for all LGPD tables
CREATE POLICY "Clinic isolation for consent records" ON lgpd_consent_records
  FOR ALL USING (clinic_id = get_current_clinic_id());

CREATE POLICY "Clinic isolation for audit records" ON lgpd_audit_records
  FOR ALL USING (clinic_id = get_current_clinic_id());

CREATE POLICY "Clinic isolation for retention records" ON lgpd_data_retention_records
  FOR ALL USING (clinic_id = get_current_clinic_id());
```
