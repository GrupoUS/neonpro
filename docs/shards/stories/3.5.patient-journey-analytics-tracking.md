## 🗃️ Database Schema Updates

### New Tables Required
```sql
-- Patient Journey Events
CREATE TABLE patient_journey_events (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  event_type VARCHAR(50) NOT NULL,
  event_category VARCHAR(50) NOT NULL,
  event_data JSONB,
  touchpoint_id UUID,
  channel VARCHAR(50),
  timestamp TIMESTAMPTZ DEFAULT NOW(),
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Journey Stages
CREATE TABLE journey_stages (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  stage_name VARCHAR(100) NOT NULL,
  stage_order INTEGER,
  started_at TIMESTAMPTZ,
  completed_at TIMESTAMPTZ,
  status VARCHAR(20) DEFAULT 'active',
  duration_minutes INTEGER,
  satisfaction_score DECIMAL(3,2),
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Touchpoints
CREATE TABLE patient_touchpoints (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  touchpoint_type VARCHAR(50) NOT NULL,
  channel VARCHAR(50) NOT NULL,
  interaction_quality DECIMAL(3,2),
  sentiment_score DECIMAL(3,2),
  outcome VARCHAR(50),
  staff_id UUID REFERENCES staff(id),
  timestamp TIMESTAMPTZ DEFAULT NOW(),
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Satisfaction Surveys
CREATE TABLE satisfaction_surveys (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  survey_type VARCHAR(50) NOT NULL,
  overall_score DECIMAL(3,2),
  service_score DECIMAL(3,2),
  facility_score DECIMAL(3,2),
  staff_score DECIMAL(3,2),
  recommendation_score INTEGER, -- NPS
  feedback_text TEXT,
  collected_at TIMESTAMPTZ DEFAULT NOW(),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Churn Predictions
CREATE TABLE churn_predictions (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  risk_score DECIMAL(3,2) NOT NULL,
  risk_level VARCHAR(20) NOT NULL, -- low, medium, high, critical
  prediction_factors JSONB,
  recommended_actions JSONB,
  predicted_at TIMESTAMPTZ DEFAULT NOW(),
  outcome_verified BOOLEAN DEFAULT FALSE,
  actual_churned BOOLEAN,
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```
