## 🏗️ Technical Implementation

### Database Schema

```sql
-- Tabela principal de mensagens
CREATE TABLE communication_messages (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    conversation_id UUID NOT NULL REFERENCES communication_conversations(id),
    sender_id UUID NOT NULL REFERENCES auth.users(id),
    content TEXT NOT NULL,
    message_type VARCHAR(50) NOT NULL DEFAULT 'text', -- text, image, file, template
    metadata JSONB DEFAULT '{}',
    read_at TIMESTAMP WITH TIME ZONE,
    delivered_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    deleted_at TIMESTAMP WITH TIME ZONE
);

-- Conversas/Threads
CREATE TABLE communication_conversations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    type VARCHAR(50) NOT NULL, -- 'internal', 'patient_chat', 'broadcast'
    title VARCHAR(255),
    participants UUID[] NOT NULL,
    patient_id UUID REFERENCES patients(id),
    clinic_id UUID NOT NULL REFERENCES clinics(id),
    metadata JSONB DEFAULT '{}',
    archived_at TIMESTAMP WITH TIME ZONE,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Templates de comunicação
CREATE TABLE communication_templates (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    clinic_id UUID NOT NULL REFERENCES clinics(id),
    name VARCHAR(255) NOT NULL,
    description TEXT,
    template_type VARCHAR(50) NOT NULL, -- 'email', 'sms', 'push', 'chat'
    subject VARCHAR(255),
    content TEXT NOT NULL,
    variables JSONB DEFAULT '[]', -- Lista de variáveis disponíveis
    triggers JSONB DEFAULT '[]', -- Eventos que ativam o template
    active BOOLEAN DEFAULT true,
    created_by UUID NOT NULL REFERENCES auth.users(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Notificações
CREATE TABLE communication_notifications (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    recipient_id UUID NOT NULL REFERENCES auth.users(id),
    type VARCHAR(50) NOT NULL, -- 'push', 'email', 'sms'
    title VARCHAR(255),
    content TEXT NOT NULL,
    data JSONB DEFAULT '{}',
    scheduled_for TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    sent_at TIMESTAMP WITH TIME ZONE,
    delivered_at TIMESTAMP WITH TIME ZONE,
    read_at TIMESTAMP WITH TIME ZONE,
    failed_at TIMESTAMP WITH TIME ZONE,
    error_message TEXT,
    clinic_id UUID NOT NULL REFERENCES clinics(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Consentimentos de comunicação
CREATE TABLE communication_consents (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    consent_type VARCHAR(50) NOT NULL, -- 'email', 'sms', 'push', 'marketing'
    consented BOOLEAN NOT NULL DEFAULT false,
    consented_at TIMESTAMP WITH TIME ZONE,
    revoked_at TIMESTAMP WITH TIME ZONE,
    ip_address INET,
    user_agent TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    updated_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    
    UNIQUE(patient_id, consent_type)
);
```

export function useCommunicationRealtime(conversationId: string) {
  const supabase = createClient();
  const [messages, setMessages] = useState<Message[]>([]);
  const [isConnected, setIsConnected] = useState(false);

  useEffect(() => {

      .channel(`conversation:${conversationId}`)

        setMessages(prev => [...prev, payload.message]);
      })
      .subscribe();

1. Implementar debounce e batching para otimizar updates

4. Paginação virtual e lazy loading
5. Índices de busca separados com metadata não-criptografada
