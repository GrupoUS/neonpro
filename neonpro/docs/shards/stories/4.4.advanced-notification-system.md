## 🎯 Business Value
**Problem Statement**: Clínicas estéticas precisam de um sistema robusto de notificações multi-canal para reduzir no-shows, melhorar comunicação com pacientes e automatizar lembretes. Atualmente, comunicação é manual e não há escalação automática para não-resposta.

**Success Criteria**:
- Sistema de lembretes automáticos multi-canal (SMS, Email, WhatsApp, Push)
- Confirmação inteligente de presença com múltiplas tentativas
- Notificações personalizadas por perfil e preferências do paciente
- Sistema de escalação automática para não-resposta
- Analytics completo de entrega e efetividade
- Integração nativa com WhatsApp Business API

**Acceptance Criteria**:

## 🗃️ Database Schema Updates

### New Tables Required
```sql
-- Notification Templates
CREATE TABLE notification_templates (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  type VARCHAR(50) NOT NULL, -- reminder, confirmation, escalation
  channel VARCHAR(20) NOT NULL, -- sms, email, whatsapp, push
  subject VARCHAR(200),
  content TEXT NOT NULL,
  variables JSONB, -- {patient_name}, {appointment_date}, etc.
  active BOOLEAN DEFAULT true,
  clinic_id UUID REFERENCES clinics(id),
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Notification Rules
CREATE TABLE notification_rules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  trigger_type VARCHAR(50) NOT NULL, -- appointment_created, reminder_time, etc.
  conditions JSONB, -- {procedure_type: "botox", hours_before: 24}
  template_id UUID REFERENCES notification_templates(id),
  channel_priority VARCHAR[] DEFAULT ARRAY['whatsapp', 'sms', 'email'],
  max_attempts INTEGER DEFAULT 3,
  retry_interval_minutes INTEGER DEFAULT 60,
  active BOOLEAN DEFAULT true,
  clinic_id UUID REFERENCES clinics(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Notification Queue
CREATE TABLE notification_queue (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id),
  appointment_id UUID REFERENCES appointments(id),
  template_id UUID REFERENCES notification_templates(id),
  channel VARCHAR(20) NOT NULL,
  recipient VARCHAR(100) NOT NULL, -- phone/email/whatsapp_id
  subject VARCHAR(200),
  content TEXT NOT NULL,
  scheduled_for TIMESTAMPTZ NOT NULL,
  status VARCHAR(20) DEFAULT 'pending', -- pending, sent, delivered, failed, cancelled
  attempts INTEGER DEFAULT 0,
  error_message TEXT,
  sent_at TIMESTAMPTZ,
  delivered_at TIMESTAMPTZ,
  read_at TIMESTAMPTZ,
  clicked_at TIMESTAMPTZ,
  replied_at TIMESTAMPTZ,
  provider_id VARCHAR(100), -- external provider message ID
  metadata JSONB,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- Notification Analytics
CREATE TABLE notification_analytics (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  date DATE NOT NULL,
  clinic_id UUID REFERENCES clinics(id),
  channel VARCHAR(20) NOT NULL,
  template_type VARCHAR(50) NOT NULL,
  total_sent INTEGER DEFAULT 0,
  total_delivered INTEGER DEFAULT 0,
  total_read INTEGER DEFAULT 0,
  total_clicked INTEGER DEFAULT 0,
  total_replied INTEGER DEFAULT 0,
  total_failed INTEGER DEFAULT 0,
  avg_delivery_time_seconds INTEGER,
  avg_read_time_seconds INTEGER,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  UNIQUE(date, clinic_id, channel, template_type)
);

-- Patient Notification Preferences
CREATE TABLE patient_notification_preferences (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  patient_id UUID REFERENCES patients(id) UNIQUE,
  preferred_channels VARCHAR[] DEFAULT ARRAY['whatsapp', 'sms'],
  preferred_time_start TIME DEFAULT '09:00',
  preferred_time_end TIME DEFAULT '18:00',
  timezone VARCHAR(50) DEFAULT 'America/Sao_Paulo',
  language VARCHAR(10) DEFAULT 'pt-BR',
  frequency_limit INTEGER DEFAULT 5, -- max notifications per day
  allow_marketing BOOLEAN DEFAULT false,
  allow_reminders BOOLEAN DEFAULT true,
  allow_confirmations BOOLEAN DEFAULT true,
  whatsapp_number VARCHAR(20),
  email VARCHAR(100),
  phone VARCHAR(20),
  last_response_channel VARCHAR(20),
  last_response_at TIMESTAMPTZ,
  response_score INTEGER DEFAULT 50, -- 0-100 based on response history
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- Escalation Rules
CREATE TABLE escalation_rules (
  id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  name VARCHAR(100) NOT NULL,
  trigger_condition VARCHAR(100) NOT NULL, -- no_response_24h, no_confirmation_4h
  target_user_id UUID REFERENCES auth.users(id),
  notification_template_id UUID REFERENCES notification_templates(id),
  delay_minutes INTEGER DEFAULT 0,
  max_escalations INTEGER DEFAULT 2,
  active BOOLEAN DEFAULT true,
  clinic_id UUID REFERENCES clinics(id),
  created_at TIMESTAMPTZ DEFAULT NOW()
);
```

### Indexes for Performance
```sql
CREATE INDEX idx_notification_queue_scheduled ON notification_queue(scheduled_for, status);
CREATE INDEX idx_notification_queue_patient ON notification_queue(patient_id);
CREATE INDEX idx_notification_queue_appointment ON notification_queue(appointment_id);
CREATE INDEX idx_notification_analytics_date_clinic ON notification_analytics(date, clinic_id);
CREATE INDEX idx_patient_preferences_patient ON patient_notification_preferences(patient_id);
```
