## 🎯 Story Description

Desenvolver um portal completo de auto-atendimento para pacientes que permita agendamento online, acesso ao histórico médico, upload de documentos, acompanhamento de progresso em tempo real e sistema de avaliações. O portal deve ser totalmente responsivo, acessível (WCAG 2.1 AA) e integrado com o sistema de comunicação já implementado.

## 🏗️ Technical Implementation

### Database Schema Additions

```sql
-- Extensões do portal do paciente
CREATE TABLE patient_portal_sessions (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    session_token VARCHAR(255) NOT NULL UNIQUE,
    expires_at TIMESTAMP WITH TIME ZONE NOT NULL,
    last_activity TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    ip_address INET,
    user_agent TEXT,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW()
);

-- Uploads de arquivos
CREATE TABLE patient_uploads (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    filename VARCHAR(255) NOT NULL,
    original_filename VARCHAR(255) NOT NULL,
    file_type VARCHAR(100) NOT NULL,
    file_size INTEGER NOT NULL,
    category VARCHAR(50) NOT NULL, -- 'exam', 'document', 'progress_photo'
    storage_path TEXT NOT NULL,
    upload_date TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    is_processed BOOLEAN DEFAULT false,
    metadata JSONB DEFAULT '{}',
    clinic_id UUID NOT NULL REFERENCES clinics(id)
);

-- Avaliações e feedback
CREATE TABLE patient_evaluations (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    appointment_id UUID REFERENCES appointments(id),
    overall_rating INTEGER CHECK (overall_rating >= 1 AND overall_rating <= 5),
    service_rating INTEGER CHECK (service_rating >= 1 AND service_rating <= 5),
    facility_rating INTEGER CHECK (facility_rating >= 1 AND facility_rating <= 5),
    staff_rating INTEGER CHECK (staff_rating >= 1 AND staff_rating <= 5),
    nps_score INTEGER CHECK (nps_score >= 0 AND nps_score <= 10),
    qualitative_feedback TEXT,
    would_recommend BOOLEAN,
    feedback_categories TEXT[], -- ['punctuality', 'cleanliness', 'communication']
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    clinic_id UUID NOT NULL REFERENCES clinics(id)
);

-- Progresso de tratamentos
CREATE TABLE treatment_progress (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    treatment_id UUID NOT NULL, -- Referência para treatments table
    session_number INTEGER NOT NULL,
    progress_percentage DECIMAL(5,2),
    notes TEXT,
    measurements JSONB DEFAULT '{}',
    photos TEXT[], -- Array de URLs das fotos
    next_session_date TIMESTAMP WITH TIME ZONE,
    created_by UUID NOT NULL REFERENCES auth.users(id),
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    clinic_id UUID NOT NULL REFERENCES clinics(id)
);

-- Agendamentos online
CREATE TABLE online_bookings (
    id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
    patient_id UUID NOT NULL REFERENCES patients(id),
    requested_date TIMESTAMP WITH TIME ZONE NOT NULL,
    requested_time TIME NOT NULL,
    service_type VARCHAR(100) NOT NULL,
    preferred_staff_id UUID REFERENCES auth.users(id),
    status VARCHAR(50) DEFAULT 'pending', -- 'pending', 'confirmed', 'cancelled'
    special_requests TEXT,
    confirmation_sent BOOLEAN DEFAULT false,
    created_at TIMESTAMP WITH TIME ZONE DEFAULT NOW(),
    clinic_id UUID NOT NULL REFERENCES clinics(id)
);
```
